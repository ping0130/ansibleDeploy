cat start.sh
docker run --name mysql \
-p 3306:3306 \
-v /data/mysql/conf.d:/etc/mysql/conf.d \
-v /data/mysql/logs:/logs \
-v /data/mysql/data:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=root123567 \
-d mysql:5.7



cat conf.d/my_custom.cnf
[mysqld]
wait_timeout=28800
interactive_timeout=28800
skip-name-resolve


cat elasticsearch.sh
#!/bin/bash

docker run -d \
    --name elasticsearch \
    --privileged \
    --restart always \
    -p 9200:9200 \
    -p 9300:9300 \
    -e "discovery.type=single-node" \
    -e "ES_JAVA_OPTS=-Xms2g -Xmx2g" \
    -v /data/elk/elasticsearch/data:/usr/share/elasticsearch/data \
    -v /data/elk/elasticsearch/elasticsearch.yml:/usr/share/elasticsearch/config/elasticsearch.yml \
    elasticsearch:8.5.0
